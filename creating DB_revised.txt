1. Users Table

CREATE TABLE `users` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `u_id` varchar(50) NOT NULL,
  `u_pw` varchar(200) NOT NULL,
  `u_bday` timestamp NOT NULL DEFAULT NOW(),
  `u_pen` int unsigned NOT NULL DEFAULT '0',
  `u_tan` int unsigned NOT NULL DEFAULT '0',
  `present_login` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `last_login` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `u_level` varchar(100) NOT NULL DEFAULT 'Kosa Kata',
  `u_haipur` int unsigned NOT NULL DEFAULT '10000',
  `sex` char(2) NOT NULL DEFAULT 'n',
  `email` varchar(255) NOT NULL,
  `vcode` varchar(10) DEFAULT NULL,
  `verification` tinyint(1) NOT NULL DEFAULT '0',
  `verify_date` timestamp NOT NULL DEFAULT 0,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

  `public` char(2) DEFAULT 'p',

  `date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,

  `com` int unsigned NOT NULL DEFAULT '0',
  `p_like` int unsigned NOT NULL DEFAULT '0',

  `score` decimal(11,2) NOT NULL DEFAULT '0.00',
